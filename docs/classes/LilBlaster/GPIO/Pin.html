def direction=(dir)
  raise ArgumentError unless %i[input output].include?(dir)
  return if  direction == dir

  @direction = dir
  gpio_pin.mode = dir == :input ? 0 : 1
end

def wait(timeout = nil)
  if timeout.nil?
    nil until on?
    true
  else
    start_time = Time.now

    timeup, pinon = nil

    loop do
      timeup = Time.now - start_time > timeout
      pinon = on?

      break if timeup || pinon
    end

    pinon
  end
end
