# File lib/lil_blaster/protocol/mark_space_encoding.rb, line 72
def int_to_pulses(int)
  binary_pad(int).chars.map do |ch|
    case ch
    when /0/
      zero_value.clone
    when /1/
      one_value.clone
    end
  end
end
