def send_code(sym, codex = nil)
  dex = codex || default_codex
  raise ArgumentError 'No codex provided' unless dex

  transmit dex.call(sym)
end

def transmit(data, args = {})
  _pin = pin unless pin
  mtd = args.fetch(:blocking, true) ? :transmit! : :transmit

  GPIO::Wave.public_send(mtd, data)
end

def default_codex
  return unless LilBlaster::ConfigFile[:default_codex]

  LilBlaster::Codex.autoload.find do |codex|
    codex.remote_name.downcase == LilBlaster::ConfigFile[:default_codex].downcase
  end
end
